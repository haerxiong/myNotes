2、安装rz、sz简单上传、下载工具

3、开启ftp功能
	yum install vsftpd -y
	修改用户权限
	vi /etc/vsftpd/ftpusers 
	vi /etc/vsftpd/user_list
	启动
	service vsftpd start

	-----------------
	服务器有防火墙时，需要设置vsftp的pasv模式端口范围，并开放此端口
	vi /etc/vsftpd/vsftpd.conf
	pasv_min_port=10000
	pasv_max_port=10010

	查看selinux安全服务的状态
		getenforce
	临时关闭服务，无需重启
		setenforce 0
	永久关闭，修改配置文件
		vi /etc/sysconfig/selinux
		改为SELINUX=disabled

8、安装jdk、配置环境变量
	linux间内网文件传输
	scp linux-jdk1.7.zip root@47.92.159.35:/home

	tar zxvf linux-jdk1.7.gz
	或者
	unzip linux-jdk1.7.zip

	mv jdk1.7.0_79/ /usr/local/

	chmod +x /usr/local/jdk1.7.0_79/bin/*
	chmod +x /usr/local/jdk1.7.0_79/jre/bin/*

	/etc/profile文件底部追加：
	vi /etc/profile

JAVA_HOME=/usr/local/jdk1.7.0_79
JRE_HOME=/usr/local/jdk1.7.0_79/jre
PATH=$JAVA_HOME/bin:$JRE_HOME/bin:$PATH
CLASSPATH=.:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar:$JRE_HOME/lib
export JAVA_HOME JRE_HOME PATH CLASSPATH

	立即生效配置
	source /etc/profile
	或
	. /etc/profile(有空格)
